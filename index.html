# hello world

# 安装

npm install io-ui

# 导入

在 main.js 中写入以下内容：

import Vue from 'vue'
import App from './App.vue'
import IoUi from 'io-ui'
import 'io-ui/package/main.css'

Vue.config.productionTip = false
Vue.use(IoUi)

new Vue({
  render: h => h(App),
}).$mount('#app')
